CREATE TABLE objek_wisata (
	peringkat INT,
	nama_objek VARCHAR(100) NOT NULL,
	nama_daerah VARCHAR(50) NOT NULL,
	rata_rata_rating FLOAT,
	jumlah_rating_5 INT,
	jumlah_rating_4 INT,
	jumlah_rating_3 INT,
	jumlah_rating_2 INT,
	jumlah_rating_1 INT,
	jumlah_ulasan INT,
	PRIMARY KEY (peringkat),
	FOREIGN KEY (nama_daerah) REFERENCES daerah(nama_daerah)
)

CREATE TABLE daerah (
	nama_daerah VARCHAR(100),
	jumlah_objek_sedaerah INT,
	PRIMARY KEY (nama_daerah)
)

CREATE TABLE pengelola (
	nama_pengelola VARCHAR(50),
	kontak_pengelola VARCHAR(12) NOT NULL,
	PRIMARY KEY (nama_pengelola)
)

CREATE TABLE jenis (
	peringkat INT,
	nama_jenis VARCHAR(100),
	PRIMARY KEY (peringkat, nama_jenis),
	FOREIGN KEY (peringkat) REFERENCES objek_wisata(peringkat)
)

CREATE TABLE operasional (
	kode_operasional CHAR(10),
	jam_buka TIME,
	jam_tutup TIME,
	hari_operasional VARCHAR(50),
	PRIMARY KEY (kode_operasional)
)	

CREATE TABLE ulasan_teratas (
	peringkat INT,
	nama_akun_pengunjung VARCHAR(100),
	asal_pengunjung VARCHAR(50),
	rating_pengunjung FLOAT,
	waktu_berkunjung VARCHAR(20),
	tipe_kunjungan VARCHAR(20),
	PRIMARY KEY (peringkat, nama_akun_pengunjung),
	FOREIGN KEY (peringkat) REFERENCES objek_wisata(peringkat)
)
